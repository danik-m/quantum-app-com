            <label>Ширина: <input type="number" id="width" value="2.0"></label>

            <button onclick="run()">Запуск</button>

        </div>

        <div id="status" style="margin-top: 10px; color: gray;"></div>

        <img id="result" src="" style="display:none;">

    <script>

        async function run() {

            const E = document.getElementById('energy').value;

            const W = document.getElementById('width').value;

            const status = document.getElementById('status');

            const img = document.getElementById('result');

            

            status.innerText = "Вычисление...";

            img.style.display = 'none';


            try {

                // Запрос к нашему API

                const response = await fetch(`/api/simulate?energy=${E}&width=${W}`);

                

                if (!response.ok) {

                    throw new Error(`Ошибка сети: ${response.status}`);

                }


                const data = await response.json();

                img.src = "data:image/png;base64," + data.image;

                img.style.display = 'block';

                status.innerText = "Готово!";

            } catch (e) {

                status.innerText = "Ошибка: " + e.message;

            }

        }

    </script>
